@st.cache_resource
def load_model():
    """Load the trained model and preprocessing objects"""
    try:
        model = tf.keras.models.load_model('ev_energy_predictor_model.h5')
        scaler = joblib.load('feature_scaler.pkl')
        label_encoders = joblib.load('label_encoders.pkl')
        
        with open('feature_names.txt', 'r') as f:
            feature_names = [line.strip() for line in f.readlines()]
        
        with open('model_metrics.json', 'r') as f:
            metrics = json.load(f)
        
        return model, scaler, label_encoders, feature_names, metrics
    except:
        return None, None, None, None, None

def calculate_physics_baseline(distance, speed, elevation, mass, drag_coeff, frontal_area):
    """Calculate physics-based energy consumption"""
    g = 9.81
    rho = 1.225
    roll_resist = 0.0095
    
    # Rolling resistance energy (kWh)
    rolling_energy = roll_resist * mass * g * distance * 1000 / 3600 / 1000
    
    # Aerodynamic drag energy (kWh)
    speed_ms = speed / 3.6
    time_hours = distance / speed
    aero_energy = 0.5 * drag_coeff * rho * frontal_area * (speed_ms ** 3) * time_hours * 3600 / 3600 / 1000
    
    # Potential energy (kWh)
    potential_energy = mass * g * elevation / 3600 / 1000
    
    return rolling_energy + aero_energy + max(potential_energy, 0)

def create_route_map(origin_city, dest_city):
    """Create an interactive map with route"""
    origin = INDIAN_CITIES[origin_city]
    dest = INDIAN_CITIES[dest_city]
    
    # Calculate center point
    center_lat = (origin['lat'] + dest['lat']) / 2
    center_lon = (origin['lon'] + dest['lon']) / 2
    
    # Create map
    m = folium.Map(location=[center_lat, center_lon], zoom_start=6)
    
    # Add markers
    folium.Marker(
        [origin['lat'], origin['lon']],
        popup=f"Origin: {origin_city}",
        icon=folium.Icon(color='green', icon='play')
    ).add_to(m)
    
    folium.Marker(
        [dest['lat'], dest['lon']],
        popup=f"Destination: {dest_city}",
        icon=folium.Icon(color='red', icon='stop')
    ).add_to(m)
    
    # Add route line
    folium.PolyLine(
        locations=[[origin['lat'], origin['lon']], [dest['lat'], dest['lon']]],
        color='blue',
        weight=3,
        opacity=0.7
    ).add_to(m)
    
    return m

def create_energy_gauge(predicted_energy, battery_capacity):
    """Create a gauge chart for energy consumption"""
    percentage = (predicted_energy / battery_capacity) * 100
    
    fig = go.Figure(go.Indicator(
        mode="gauge+number+delta",
        value=predicted_energy,
        domain={'x': [0, 1], 'y': [0, 1]},
        title={'text': "Energy Consumption (kWh)", 'font': {'size': 20}},
        delta={'reference': battery_capacity * 0.3},
        gauge={
            'axis': {'range': [None, battery_capacity]},
            'bar': {'color': "darkblue"},
            'steps': [
                {'range': [0, battery_capacity * 0.3], 'color': "lightgreen"},
                {'range': [battery_capacity * 0.3, battery_capacity * 0.7], 'color': "yellow"},
                {'range': [battery_capacity * 0.7, battery_capacity], 'color': "red"}
            ],
            'threshold': {
                'line': {'color': "red", 'width': 4},
                'thickness': 0.75,
                'value': battery_capacity * 0.9
            }
        }
    ))
    
    fig.update_layout(height=300)
    return fig
